142 template <typename T> std::vector<T> getRepeatableOptionArg(const std::string &argName) const
143 {
144 auto it = optionArgs->find(argName);
145 if (it == optionArgs->end())
146 throw MissingArgumentException(argName, *optionArgs);
147
148 try
149 {
150 auto anyVec = getAnyCast<std::vector<std::any>>(argName, *optionArgs);
151 std::vector<T> result;
152 result.reserve(anyVec.size());
153
154 for (const auto &elem : anyVec)
155 {
156 result.push_back(std::any_cast<T>(elem));
157 }
158
159 return result;
160 }
161 catch (const std::bad_any_cast &)
162 {
163 throw InvalidArgumentTypeException(argName, typeid(std::vector<T>), it->second.type());
164 }
165 }

202 template <typename T> T getArg(const std::string &argName) const
203 {
204 if (isPositionalArgPresent(argName))
205 {
206 return getAnyCast<T>(argName, *positionalArgs);
207 }
208 else if (isOptionArgPresent(argName))
209 {
210 return getAnyCast<T>(argName, *optionArgs);
211 }
212 else
213 {
214 throw MissingArgumentException(argName, *positionalArgs);
215 }
216 }

222 template <typename T> auto getRepeatableArg(const std::string &argName) const
223 {
224 if (isPositionalArgPresent(argName))
225 {
226 return getRepeatablePositionalArg<T>(argName);
227 }
228 else if (isOptionArgPresent(argName))
229 {
230 return getRepeatableOptionArg<T>(argName);
231 }
232 else
233 {
234 throw MissingArgumentException(argName, *positionalArgs);
235 }
236 }

245 template <typename T>
246 static T getAnyCast(const std::string &name, std::unordered_map<std::string, std::any> &dict)
247 {
248 try
249 {
250 auto it = dict.find(name);
251 if (it == dict.end())
252 {
253 throw MissingArgumentException(name, dict);
254 }
255 return std::any_cast<T>(it->second);
256 }
257 catch (const std::bad_any_cast &)
258 {
259 throw InvalidArgumentTypeException(name, typeid(T), dict.at(name).type());
260 }
261 }
